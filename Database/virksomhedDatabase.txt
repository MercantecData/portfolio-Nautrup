DELIMITER $$
CREATE PROCEDURE GetNameAndJobTitel()
BEGIN
SELECT CONCAT (employee.first_name, ' ', employee.last_name) As Navn, titles.title AS JobTitel FROM employee 
INNER JOIN titles 
ON employee.id = titles.employee_Id ;
End $$
DELIMITER ;

/* */

DELIMITER $$

CREATE PROCEDURE GetNameAndDepartment()
BEGIN
	SELECT CONCAT (employee.first_name,' ', employee.last_name) AS Navn, department.dep_name FROM employee
    INNER JOIN dep_employee
    ON employee.id = dep_employee.id ;
END $$

DELIMITER ;
